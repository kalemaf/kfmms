GL MAPPING EXAMPLE:

$gl = new GLMapping($c);

// Map equipment to GL account
$gl->mapEquipmentToAccount(42, '61200', 'MAINT-001');

// Create GL entry on WO completion
$je = $gl->createWOJournalEntry(123, 1500.00);
// Returns: journal_entry_id, debit_account, credit_account

// Post to GL
$gl->postJournalEntry($je['journal_entry_id']);

// Split costs across accounts
$gl->allocateWOCost(123, [
    ['account' => '61000', 'percentage' => 60],
    ['account' => '61500', 'percentage' => 40]
]);

// Get account balance
$balance = $gl->getAccountBalance('61000', '2026-01-01', '2026-03-31');
// Returns: debits, credits, balance
